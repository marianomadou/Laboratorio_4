SELECT prestamo.id AS 'ID', giles.nombre AS 'Nombre Empleado', equipo.nombre AS "equipo", edif1.nombre AS "edifi equipo", edif2.nombre AS 'edificio Empleado' FROM bienes AS equipo, empleados AS giles, edificios AS edif1, edificios AS edif2, empleadouso AS prestamo WHERE prestamo.idEmpleados = giles.id AND prestamo.idBienes = equipo.id AND edif1.id = equipo.edificio AND giles.edificio = edif2.id
